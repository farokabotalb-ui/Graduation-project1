    <script>
        // دالة عرض الصفحة النشطة - تم تعديلها لمنع التداخل
        function showPage(pageId) {
            // إخفاء كل الصفحات بشكل صريح
            document.querySelectorAll('.page').forEach(p => {
                p.style.display = 'none';
                p.classList.remove('active');
            });
            
            // إظهار الصفحة المطلوبة
            const targetPage = document.getElementById(pageId);
            targetPage.style.display = 'block';
            targetPage.classList.add('active');
            
            // تحديث الروابط النشطة
            document.querySelectorAll('.navbar a').forEach(l => l.classList.remove('active'));
            document.getElementById('nav-' + pageId).classList.add('active');
            
            // إظهار زر الطباعة المناسب للصفحة النشطة
            document.querySelectorAll('.print-btn-group').forEach(btn => btn.style.display = 'none');
            
            if (pageId === 'page1') {
                document.getElementById('print-btn-page1').style.display = 'flex';
            } else if (pageId === 'page2') {
                document.getElementById('print-btn-page2').style.display = 'flex';
            } else if (pageId === 'page3') {
                document.getElementById('print-btn-page3').style.display = 'flex';
            }
        }

        // دالة طباعة صفحة محددة فقط
        function printSpecificPage(pageId) {
            try {
                // حفظ جميع الصفحات
                const allPages = document.querySelectorAll('.page');
                const printButtons = document.querySelector('.print-section-buttons');
                const navbar = document.querySelector('.navbar');
                const footer = document.querySelector('footer');
                
                // إخفاء العناصر غير المراد طباعتها
                allPages.forEach(p => {
                    p.style.display = 'none';
                });
                printButtons.style.display = 'none';
                navbar.style.display = 'none';
                if (footer) footer.style.display = 'none';
                
                // إظهار الصفحة المطلوبة فقط
                const pageToPrint = document.getElementById(pageId);
                pageToPrint.style.display = 'block';
                pageToPrint.classList.add('active');
                
                // تأخير بسيط للتأكد من تطبيق الأنماط
                setTimeout(function() {
                    // طباعة الصفحة
                    window.print();
                }, 100);
                
                // استعادة العرض الطبيعي بعد الطباعة
                setTimeout(function() {
                    // إخفاء كل الصفحات أولاً
                    allPages.forEach(p => {
                        p.style.display = 'none';
                        p.classList.remove('active');
                    });
                    
                    // إظهار الصفحة النشطة الأصلية (الصفحة الأولى كإفتراضي)
                    const defaultPage = document.getElementById('page1');
                    defaultPage.style.display = 'block';
                    defaultPage.classList.add('active');
                    
                    // إظهار العناصر الأخرى
                    printButtons.style.display = 'flex';
                    navbar.style.display = '';
                    if (footer) footer.style.display = '';
                    
                    // تحديث أزرار الطباعة
                    document.querySelectorAll('.print-btn-group').forEach(btn => btn.style.display = 'none');
                    document.getElementById('print-btn-page1').style.display = 'flex';
                    
                    // تحديث الرابط النشط
                    document.querySelectorAll('.navbar a').forEach(l => l.classList.remove('active'));
                    document.getElementById('nav-page1').classList.add('active');
                    
                }, 1000);
                
            } catch (error) {
                alert('حدث خطأ أثناء الطباعة: ' + error.message);
                // استعادة العرض الطبيعي في حالة الخطأ
                location.reload();
            }
        }

        // منع إعادة تحميل الصفحة عند النقر على الروابط
        document.querySelectorAll('.navbar a').forEach(link => {
            link.addEventListener('click', e => e.preventDefault());
        });

        // تهيئة الصفحة: التأكد من أن الصفحة الأولى فقط هي الظاهرة
        window.addEventListener('load', function() {
            // إخفاء كل الصفحات أولاً
            document.querySelectorAll('.page').forEach(p => {
                p.style.display = 'none';
                p.classList.remove('active');
            });
            
            // إظهار الصفحة الأولى فقط
            const page1 = document.getElementById('page1');
            page1.style.display = 'block';
            page1.classList.add('active');
            
            // إظهار زر الطباعة للصفحة الأولى
            document.getElementById('print-btn-page1').style.display = 'flex';
            document.getElementById('print-btn-page2').style.display = 'none';
            document.getElementById('print-btn-page3').style.display = 'none';
            
            console.log('مشروع التخرج جاهز - النسخة الموسعة 2026');
            console.log('تم إصلاح مشكلة تداخل الصفحات');
        });
    </script>
